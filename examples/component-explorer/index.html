<ai-insight-card>

The flagship component of Cognivo. Displays data with AI-powered insights, explanations, forecasts, and anomaly detection built-in.

üì¶ Installation

pnpm add @cognivo/components @cognivo/adapter-openai

üíª Basic Usage

<script type="module">
  import '@cognivo/components';
  import { OpenAiClient } from '@cognivo/adapter-openai';

  const aiClient = new OpenAiClient({
    apiKey: 'YOUR_API_KEY'
  });

  const card = document.querySelector('ai-insight-card');
  card.data = [
    { month: 'Jan', revenue: 10000 },
    { month: 'Feb', revenue: 12000 },
    { month: 'Mar', revenue: 15000 }
  ];
  card.aiClient = aiClient;
  card.meta = {
    timeframe: 'monthly',
    unit: 'USD',
    dataType: 'revenue'
  };
</script>

<ai-insight-card
  title="Revenue Analysis"
  ai-actions='["explain", "forecast"]'
>
  <div>Your chart here</div>
</ai-insight-card>

Prop Type Default Description
